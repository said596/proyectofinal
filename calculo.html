Cálculo de Magnitudes Matemáticas o Físicas

Teoría

  • Área de un círculo: Área = πr2
  • Volumen de una esfera: Volumen = 4/3 πr3
  • Velocidad: Velocidad = Distancia / Tiempo
  • Fuerza: Fuerza = m × a
  • Energía cinética: Energía cinética = 1/2 m v2

Diagrama de Flujo

Ejemplo de Código en C

#include <stdio.h> #include <math.h> // Constante Pi #define PI 3.14159 // Funciones para el cálculo de las magnitudes float calcularAreaCirculo(float radio) { return PI * radio * radio; } float calcularVolumenEsfera(float radio) { return (4.0/3.0) * PI * pow(radio, 3); } float calcularVelocidad(float distancia, float tiempo) { return distancia / tiempo; } float calcularFuerza(float masa, float aceleracion) { return masa * aceleracion; } float calcularEnergiaCinetica(float masa, float velocidad) { return 0.5 * masa * pow(velocidad, 2); } int main() { float radio, distancia, tiempo, masa, aceleracion, velocidad; // Cálculo del área de un círculo printf("Ingrese el radio del círculo: "); scanf("%f", &radio); printf("El área del círculo es: %.2f\n", calcularAreaCirculo(radio)); // Cálculo del volumen de una esfera printf("Ingrese el radio de la esfera: "); scanf("%f", &radio); printf("El volumen de la esfera es: %.2f\n", calcularVolumenEsfera(radio)); // Cálculo de la velocidad printf("Ingrese la distancia recorrida: "); scanf("%f", &distancia); printf("Ingrese el tiempo empleado: "); scanf("%f", &tiempo); printf("La velocidad es: %.2f\n", calcularVelocidad(distancia, tiempo)); // Cálculo de la fuerza printf("Ingrese la masa: "); scanf("%f", &masa); printf("Ingrese la aceleración: "); scanf("%f", &aceleracion); printf("La fuerza es: %.2f\n", calcularFuerza(masa, aceleracion)); // Cálculo de la energía cinética printf("Ingrese la masa: "); scanf("%f", &masa); printf("Ingrese la velocidad: "); scanf("%f", &velocidad); printf("La energía cinética es: %.2f\n", calcularEnergiaCinetica(masa, velocidad)); return 0; }
